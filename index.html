// Garantir que o código só rode na página de obrigado if (window.location.pathname.includes("/obrigado")) { // Função para gerar número de protocolo aleatório function gerarProtocolo() { return Math.floor(10000 + Math.random() * 90000); // Número entre 10000 e 99999 } // Função para capturar UTMs da URL function getUTMParams() { let params = new URLSearchParams(window.location.search); return { utm_source: params.get("utm_source") || "desconhecido", utm_medium: params.get("utm_medium") || "desconhecido", utm_campaign: params.get("utm_campaign") || "desconhecido", utm_term: params.get("utm_term") || "", utm_content: params.get("utm_content") || "" }; } // Esperar a página carregar totalmente antes de modificar o botão window.addEventListener("load", function () { setTimeout(function () { var numeroWhatsApp = "5511916792333"; // Seu número do WhatsApp var protocolo = gerarProtocolo(); var utms = getUTMParams(); // Criar mensagem personalizada com protocolo e UTMs var mensagem = `Você será transferido para atendimento. Guarde este número de protocolo, pois será necessário informá-lo para continuar com seu suporte.\n\n📌 *PROTOCOLO:* ${protocolo}\n\n*Atenção:* Não apague esta mensagem para garantir um atendimento mais rápido!`; // Criar o link do WhatsApp com UTMs var linkWhatsApp = `https://wa.me/${numeroWhatsApp}?text=${encodeURIComponent(mensagem)}` + `&utm_source=${utms.utm_source}&utm_medium=${utms.utm_medium}&utm_campaign=${utms.utm_campaign}&utm_term=${utms.utm_term}&utm_content=${utms.utm_content}`; // Buscar botões que podem ser do WhatsApp var botoesExistentes = document.querySelectorAll("a.ub-button, a[href='#']"); let botaoAtualizado = false; botoesExistentes.forEach(function (botao) { if (botao.innerText.includes("WhatsApp") || botao.textContent.includes("WhatsApp")) { botao.setAttribute("href", linkWhatsApp); botaoAtualizado = true; console.log("✅ Botão do WhatsApp atualizado com sucesso:", linkWhatsApp); } }); if (!botaoAtualizado) { console.error("❌ Nenhum botão do WhatsApp foi encontrado ou atualizado!"); } }, 2000); // Tempo de espera para garantir que o Unbounce carregou tudo }); }
